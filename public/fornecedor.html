// ============================================================ // CARREGAR LISTA DE ARQUIVOS DO S3 (FORNECEDOR) // ============================================================ async function loadMediaFromS3() { try { // 1. Carregar fotos dinamicamente do S3 (SEM O BUCKET NA URL) console.log('📡 Buscando lista de fotos no S3 (fornecedor)...'); const fotosResponse = await fetch(`${API_BASE_URL}/api/gallery/list/fotos`); if (!fotosResponse.ok) { console.error('❌ Erro ao buscar fotos:', fotosResponse.status); throw new Error('Erro ao buscar lista de fotos'); } const fotosData = await fotosResponse.json(); console.log(`📸 Encontradas ${fotosData.count} fotos na pasta fotos/`); // Filtra apenas arquivos .jpg const photoFiles = fotosData.files.filter(file => file.filename.endsWith('.jpg') || file.filename.endsWith('.jpeg') ); const fotos = photoFiles.map((file, index) => { const filename = file.filename; // Tenta extrair número do nome do arquivo const numMatch = filename.match(/(\d+)\./); const num = numMatch ? numMatch[1] : String(index + 1).padStart(2, '0'); return { id: `foto_${num}`, type: 'foto', key: `fotos/${filename}`, price: PHOTO_PRICE, title: `Foto ${parseInt(num)}`, thumbnail: `${API_BASE_URL}/api/gallery/view/fotos/${filename}`, mediaUrl: `${API_BASE_URL}/api/gallery/view/fotos/${filename}`, isVideo: false }; }); // 2. Carregar vídeos dinamicamente do S3 (SEM O BUCKET NA URL) console.log('📡 Buscando lista de vídeos no S3 (fornecedor)...'); const videosResponse = await fetch(`${API_BASE_URL}/api/gallery/list/videos`); if (!videosResponse.ok) { console.error('❌ Erro ao buscar vídeos:', videosResponse.status); throw new Error('Erro ao buscar lista de vídeos'); } const videosData = await videosResponse.json(); console.log(`📹 Encontrados ${videosData.count} vídeos na pasta videos/`); // Filtra apenas arquivos .mp4 const videoFiles = videosData.files.filter(file => file.filename.endsWith('.mp4') ); // Cria placeholder SVG para vídeos function createVideoPlaceholder(title) { const svg = ` `; return `data:image/svg+xml;charset=utf-8,${encodeURIComponent(svg)}`; } const videos = videoFiles.map((file, index) => { const filename = file.filename; const numMatch = filename.match(/(\d+)\.mp4/); const num = numMatch ? numMatch[1] : String(index + 1).padStart(2, '0'); const title = `Vídeo ${parseInt(num)}`; // Usa a rota de thumbnail do backend const thumbnailUrl = `${API_BASE_URL}/api/gallery/thumbnail/videos/${filename}`; return { id: `video_${num}`, type: 'video', key: `videos/${filename}`, price: VIDEO_PRICE, title: title, thumbnail: thumbnailUrl, fallbackThumbnail: createVideoPlaceholder(title), mediaUrl: `${API_BASE_URL}/api/gallery/view/videos/${filename}`, isVideo: true }; }); // 3. Combinar tudo allMedia = [...fotos, ...videos]; totalItems = allMedia.length; console.log(`📊 Total: ${fotos.length} fotos + ${videos.length} vídeos = ${totalItems} itens`); // 4. Atualizar contadores na UI updateCounters(); // 5. Renderizar galeria renderGallery(); } catch (error) { console.error('❌ Erro ao carregar mídias:', error); alert('Erro ao carregar a galeria. Tente novamente.'); } }
